\maketitle

The \pkg{biblatex-academic} package provides a \pkg{biblatex} style for lists of things an academic might like to list in a CV or lab whitepaper (students, alumni, collaborators, proposals, projects, courses, etc.).
This implementation follows standard \pkg{biblatex} conventions, and can be used simply by loading \pkg{biblatex} with the appropriate options:
\begin{verbatim}
    \usepackage[backend=biber,bibstyle=academic]{biblatex}
\end{verbatim}
The \pkg{biblatex-academic} style requires \pkg{biblatex} 2.7a or later, requires \pkg{Biber} as the database back-end, and the following packages/styles: \pkg{biblatex-ieee}, \pkg{graphicx}, \pkg{siunitx}, \pkg{calc}, \pkg{hyperref}, and \pkg{wrapfig}.

\section{Entry Types}\label{sec:entry-types}

\pkg{biblatex-academic} defines several new entry types: 
\entry{alumnus}, 
\entry{collaborator}, 
\entry{committee},
\entry{course}, 
\entry{presentation}, 
\entry{project}, 
\entry{proposal}, 
\entry{seniordesign}, 
\entry{service},
\entry{student}. 

\begin{itemize}
  \item[\entry{alumnus}] An alum of a lab, university, etc.

    Required fields:
    \field{name},
    \field{image},
    \field{degree},
    \field{date},
    \field{thesis}

    Optional fields:
    \field{coadvisor},
    \field{coadvisor\_url},
    \field{employer},
    \field{employer\_url}

  \item[\entry{collaborator}] A research collaborator.

    Required fields:
    \field{name},
    \field{image},
    \field{position},
    \field{employer},
    \field{type}

    Optional fields:
    \field{employer\_url},
    \field{email},
    \field{website},
    \field{website\_alt}

  \item[\entry{committee}] A committee served on.
    
    Required fields:
    \field{title},
    \field{date},
    \field{student},
    \field{advisor},
    \field{chair},
    \field{member},
    \field{type}

    Optional fields:
    \field{coadvisor}

  \item[\entry{course}] A course course taught at a university.
    
    Required fields:
    \field{number},
    \field{title},
    \field{term},
    \field{academic\_year},
    \field{level},
    \field{enrollment},
    \field{instr\_mean},
    \field{instr\_std},
    \field{student\_mean},
    \field{student\_std}

    Optional fields:

  \item[\entry{presentation}] A presentation.
    Required fields:
    \field{presenter},
    \field{title},
    \field{eventtype},
    \field{event},
    \field{location},
    \field{date}

    Optional fields:

  \item[\entry{project}] A research project.
    
    Required fields:
    \field{title},
    \field{program},
    \field{investigators\_p},
    \field{duration},
    \field{amount},
    \field{status}

    Optional fields:
    \field{award\_num},
    \field{award\_url},
    \field{investigators\_co},
    \field{note}
  
  \item[\entry{proposal}] A research proposal.
    
    Required fields:
    \field{title},
    \field{program},
    \field{investigators\_p},
    \field{date},
    \field{amount},
    \field{status}

    Optional fields:
    \field{investigators\_co},
    \field{note}

  \item[\entry{seniordesign}] A senior design project.
    
    Required fields:
    \field{year},
    \field{title},
    \field{team},
    \field{advisors},
    \field{students}

    Optional fields:
    \field{note}

  \item[\entry{service}] A self-contained acti of service to a professional community.

    Required fields:
    \field{type},
    \field{conf\_name},
    \field{location},
    \field{duration}

    Optional fields:
    \field{conf\_url},
    \field{note}

  \item[\entry{student}] A student under advisement.
 
    Required fields:
    \field{name},
    \field{image},
    \field{degree},

    Optional fields:
    \field{email},
    \field{website},
    \field{website\_alt},
    \field{coadvisor},
    \field{coadvisor\_url},
    \field{bio}
\end{itemize}

\subsection{Collaborators}\label{sec:collaborators}

% Construct collaborator filters
\defbibfilter{IntCollab}{%
  type=collaborator and subtype=internal
}
\defbibfilter{ExtCollab}{%
  type=collaborator and subtype=external
}

\begin{refsection}
  \setlength{\bibitemsep}{\baselineskip}
  \nocite{*}

  \subsubsection{Internal}\label{sec:people-collaborators-internal}

  \begin{multicols}{2}
    \raggedcolumns
    \printbibliography[heading=none,resetnumbers=true,omitnumbers=true,env=people,filter=IntCollab]
  \end{multicols}
  
  \subsubsection{External}\label{sec:people-collaborators-external}
  
  \begin{multicols}{2}
    \raggedcolumns
    \printbibliography[heading=none,resetnumbers=true,omitnumbers=true,env=people,filter=ExtCollab]

\section{Service}\label{sec:service}

\defbibfilter{service-oc}{%
  type=service and subtype=conference-organizing-committee%
}
\defbibfilter{service-tpc}{%
  type=service and subtype=conference-technical-program-committee%
}
\defbibfilter{service-other}{%
  type=service and not subtype=conference-organizing-committee and not subtype=conference-technical-program-committee%
}

\begin{refsection}
  \setlength{\bibitemsep}{0in}
  \nocite{*}
  \printbibliography[heading=subbibnumbered,title=Conference Organizing Committees,filter=service-oc,resetnumbers=true,sorting=eventdate]
  \printbibliography[heading=subbibnumbered,title=Conference Technical Program Committees,filter=service-tpc,resetnumbers=true,sorting=eventdate]
  \printbibliography[heading=subbibnumbered,title=Other,notsubtype=conference-technical-program-committee,filter=service-other,resetnumbers=true,sorting=eventdate]
\end{refsection}
